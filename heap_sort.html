📚 介紹與原理

堆積排序法將陣列視為一棵「完全二元樹」。它會先將陣列調整成「最大堆積（Max-Heap）」（父節點恆大於子節點）， 此時陣列的第一個元素就是最大值。接著將最大值與陣列最後一個元素交換，並縮小範圍，再次調整剩餘元素，重複直到排序完成。

核心步驟：
1. Build Max-Heap：從最後一個非葉節點開始，向上整理，建立最大堆積。
2. Extract Max：將根節點（最大值）與目前未排序的最後一個節點交換。
3. Heapify：縮小未排序範圍，將新的根節點向下沉降，恢復最大堆積結構。

💻 JavaScript 核心實作

function heapify(arr, n, i) { let largest = i; let left = 2 * i + 1; let right = 2 * i + 2; // 尋找父、左子、右子中的最大值 if (left < n && arr[left] > arr[largest]) largest = left; if (right < n && arr[right] > arr[largest]) largest = right; // 若最大值不是原父節點，則交換並遞迴調整 if (largest !== i) { let temp = arr[i]; arr[i] = arr[largest]; arr[largest] = temp; heapify(arr, n, largest); } } function heapSort(arr) { let n = arr.length; // 1. 建立最大堆積 for (let i = Math.floor(n / 2) - 1; i >= 0; i--) heapify(arr, n, i); // 2. 依序提取最大值到陣列尾端 for (let i = n - 1; i > 0; i--) { let temp = arr[0]; arr[0] = arr[i]; arr[i] = temp; heapify(arr, i, 0); } return arr; }
